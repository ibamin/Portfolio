#define _CRT_SECURE_NO_WARNINGS //scanf_s를 scanf로 사용하기위한 전처리

# : 전처리기에 선언 (컴파일러에게 전달)
include : 뒤에 있는 라이브러리 포함선언 키워드(예약어)
define : 미리 선언하는 매크로함수 키워드(예약어)
사용 예시
#include <stdio.h>
※헤더 파일을 사용하는 이유 : 내가 쓰고자하는 라이브러리/함수 가 선언되어 있기때문에

int : 반환형
main() : 컴파일러가 컴파일할 때 가장 먼저 찾는함수 / 함수 이름 뒤에 무조건 ()가 나오고 ()안에 매개 변수가 들어감
※main이라는 함수는 단1개만 존재해야한다.
\n : 줄바꿈
/* */ : /* 주석내용 */
// : // 이후 한줄 주석내용
괄호의 종류
{ } : 복합문을 묶을때 사용
<>: 헤더/라이브러리 선언시 사용
( ) : 수식에 쓰이면 연산자의 우선순위를 정리 / 이름 뒤에 선언하면 함수가됨
[ ] : 배열의 크기/길이를 나타낼 때 사용 
사용 예시
int main(){

}

사용 예시 통합
#include <stdio.h>

int main(){
printf("Hellow world!"); //{ }로 추가 설명이 있지 않은 함수는 마지막에 ;(세미콜론)을 넣어줘야 마무리된다.
//하고 싶은 코드 작성
return 0; // main 함수에 선언한 반환형 때문에 깔끔한 마무리를 위해 넣어야함 하지만 안 넣어도 상관없음.
}

예제 코드 1
#include <stdio.h> //stdio.h라는 헤더를 컴파일러에게 포함시켜라

int main(void){ //필수 컴파일러가 가장먼저 확인하는 Main함수이기 때문에 꼭 들어가야한다. 운영체제만 호출가능
int x,y,sum; 
/* 
int(정수 자료형, 4byte),float(실수 자료형, 2byte),double(좀더 큰 실수 자료형, 8byte),char(문자형 자료형, 1byte)
수는 변수와 상수가 있다
상수 : 프로그램을 실행시키기 위해 기존에 정해놓은 변치 않는 수
변수 : 프로그램이 실행되는동안 값이 프로그램의 진행에 따라 변동되는 값을 지니는 수
변수 선언시 반드시 해야하는 것
1. 자료형선언
2. 초기화
변수이름으로 사용할 수 있는것
1.예약어가 아니한것
2.숫자로 시작하지 아니한것
3.특수기호 들어가지 아니한것
*/
x=100;
y=200;

sum=x+y;
printf("두수의 합: %d\n",sum);
/*
%d(%decimal) : c에서 10진수를 표현할 때 사용하는 형식 지정자 ppt 193p확인
*/
return 0;
} 

예제 코드 2
#include <stdio.h>

int main(){
int x,y,sum;
scanf("%d %d",&x,&y);
/*
scanf는 " "안에 형식 지정자를 사용함 또한 , 뒤에 입력받을 변수의 주소값을 전달해야하기 때문에 변수이름 앞에 &주소지정자를
적어주어야한다.
scanf라는 함수가 키보드로부터 입력값을 받아 변수로 전달하기위해 변수의 집주소를 알아야 하기때문에 변수의 주소를 알려주어야함
*/
sum=x+y;
printf("두수의 합 %d", sum);
}


사용자가 만들어 쓰는 함수 : 사용자지정 함수
#include를 통해 가져와서 쓰는 함수 : 라이브러리 함수
